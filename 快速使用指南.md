# 图标上报接收端 - 快速使用指南

## 🚀 方案一: 超简单 PHP 版本 (推荐新手)

### 特点
- ✅ 无需数据库
- ✅ 无需安装依赖
- ✅ 一个文件搞定
- ✅ 5分钟部署完成

### 步骤

#### 1. 创建文件 `report.php`

```php
<?php
// 图标上报接收端 - 超简单版本
header('Content-Type: application/json; charset=utf-8');
header('Access-Control-Allow-Origin: *');
header('Access-Control-Allow-Methods: POST');

// 数据存储目录
$dataDir = __DIR__ . '/icon_reports';
if (!is_dir($dataDir)) {
    mkdir($dataDir, 0755, true);
}

// 只处理 POST 请求
if ($_SERVER['REQUEST_METHOD'] !== 'POST') {
    http_response_code(405);
    die(json_encode(['ok' => false, 'message' => 'Method not allowed']));
}

// 获取数据
$input = file_get_contents('php://input');
$data = json_decode($input, true) ?: $_POST;

$package = $data['package'] ?? '';
$comment = $data['comment'] ?? '';
$timestamp = $data['timestamp'] ?? time();
$deviceInfo = $data['device_info'] ?? [];

// 验证包名
if (empty($package)) {
    http_response_code(400);
    die(json_encode(['ok' => false, 'message' => '缺少包名参数']));
}

// 保存数据
$report = [
    'package' => $package,
    'comment' => $comment,
    'timestamp' => $timestamp,
    'device_info' => $deviceInfo,
    'received_at' => date('Y-m-d H:i:s'),
    'ip' => $_SERVER['REMOTE_ADDR'] ?? 'unknown'
];

$filename = sprintf(
    '%s/%s_%s.json',
    $dataDir,
    date('Ymd_His'),
    preg_replace('/[^a-z0-9\-_]/i', '', $package)
);

file_put_contents($filename, json_encode($report, JSON_PRETTY_PRINT | JSON_UNESCAPED_UNICODE));

// 记录到日志文件
$logFile = $dataDir . '/reports.log';
$logLine = sprintf(
    "[%s] %s - %s - %s\n",
    date('Y-m-d H:i:s'),
    $_SERVER['REMOTE_ADDR'],
    $package,
    $comment ?: '(无描述)'
);
file_put_contents($logFile, $logLine, FILE_APPEND);

// 返回成功
echo json_encode([
    'ok' => true,
    'success' => true,
    'message' => '上报成功',
    'package' => $package
], JSON_UNESCAPED_UNICODE);
?>
```

```bash

chown www-data:www-data /var/www/html/api/report/*.php
```

#### 4. Nginx 配置

```nginx
server {
    listen 80;
    server_name plugin.vumstar.com;
    
    root /var/www/html;
    index index.php;
    
    location /api/report/icon {
        rewrite ^/api/report/icon$ /api/report/report.php last;
    }
    
    location ~ \.php$ {
        fastcgi_pass unix:/var/run/php/php7.4-fpm.sock;
        fastcgi_index index.php;
        include fastcgi_params;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
    }
}
```

## 🚀 方案二: Node.js 版本 (功能更强)

### 特点
- ✅ 支持 SQLite 数据库
- ✅ 自动统计分析
- ✅ RESTful API
- ✅ 易于扩展

### 步骤

#### 1. 安装依赖

```bash
mkdir icon-report-api
cd icon-report-api
npm init -y
npm install express better-sqlite3 cors
```

#### 2. 创建 `server.js`

```javascript
const express = require('express');
const Database = require('better-sqlite3');
const cors = require('cors');
const path = require('path');

const app = express();
const PORT = 3000;

// 中间件
app.use(cors());
app.use(express.json());
app.use(express.urlencoded({ extended: true }));

// 初始化数据库
const db = new Database('reports.db');
db.exec(`
  CREATE TABLE IF NOT EXISTS reports (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    package TEXT NOT NULL,
    comment TEXT,
    timestamp INTEGER,
    hostname TEXT,
    model TEXT,
    system TEXT,
    ip TEXT,
    user_agent TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
  );
  CREATE INDEX IF NOT EXISTS idx_package ON reports(package);
  CREATE INDEX IF NOT EXISTS idx_created_at ON reports(created_at);
`);

// 上报接口
app.post('/api/report/icon', (req, res) => {
  try {
    const { package: pkg, comment, timestamp, device_info } = req.body;
    
    if (!pkg) {
      return res.status(400).json({ ok: false, message: '缺少包名参数' });
    }
    
    const stmt = db.prepare(`
      INSERT INTO reports (package, comment, timestamp, hostname, model, system, ip, user_agent)
      VALUES (?, ?, ?, ?, ?, ?, ?, ?)
    `);
    
    const info = stmt.run(
      pkg,
      comment || '',
      timestamp || Date.now(),
      device_info?.hostname || '',
      device_info?.model || '',
      device_info?.system || '',
      req.ip,
      req.get('user-agent') || ''
    );
    
    console.log(`[${new Date().toISOString()}] Report received: ${pkg} from ${req.ip}`);
    
    res.json({
      ok: true,
      success: true,
      message: '上报成功',
      report_id: info.lastInsertRowid
    });
    
  } catch (err) {
    console.error('Error:', err);
    res.status(500).json({ ok: false, message: '服务器错误' });
  }
});

// 查询接口
app.get('/api/reports', (req, res) => {
  const { limit = 50, offset = 0, package: pkg } = req.query;
  
  let sql = 'SELECT * FROM reports';
  let params = [];
  
  if (pkg) {
    sql += ' WHERE package = ?';
    params.push(pkg);
  }
  
  sql += ' ORDER BY created_at DESC LIMIT ? OFFSET ?';
  params.push(parseInt(limit), parseInt(offset));
  
  const reports = db.prepare(sql).all(...params);
  const total = db.prepare('SELECT COUNT(*) as count FROM reports').get();
  
  res.json({
    ok: true,
    reports,
    total: total.count,
    limit: parseInt(limit),
    offset: parseInt(offset)
  });
});

// 统计接口
app.get('/api/reports/stats', (req, res) => {
  const topPackages = db.prepare(`
    SELECT package, COUNT(*) as count, MAX(created_at) as last_report
    FROM reports
    GROUP BY package
    ORDER BY count DESC
    LIMIT 20
  `).all();
  
  const total = db.prepare('SELECT COUNT(*) as count FROM reports').get();
  const today = db.prepare(`
    SELECT COUNT(*) as count FROM reports 
    WHERE DATE(created_at) = DATE('now')
  `).get();
  
  res.json({
    ok: true,
    total: total.count,
    today: today.count,
    top_packages: topPackages
  });
});

// 启动服务
app.listen(PORT, () => {
  console.log(`✅ Icon Report API running on http://localhost:${PORT}`);
  console.log(`📝 POST /api/report/icon - 接收上报`);
  console.log(`📊 GET  /api/reports - 查询记录`);
  console.log(`📈 GET  /api/reports/stats - 统计数据`);
});
```

#### 3. 启动服务

```bash
# 开发环境
node server.js

# 生产环境 (使用 PM2)
npm install -g pm2
pm2 start server.js --name icon-report-api
pm2 save
pm2 startup
```

#### 4. Nginx 反向代理

```nginx

    proxy_set_header X-Real-IP $remote_addr;
```

## 🚀 方案三: 使用在线服务 (零部署)

3. 使用 REST API 接收上报
